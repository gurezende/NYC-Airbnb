# Extracting the column Name (with qualities from the property) and putting into a list for counting
reps = []
c=0
while c < len(df3):
    try:
        strg = df3['name'][c].split()
    except:
        c=c+1
    for i in strg:
        reps.append(i)
    c=c+1
